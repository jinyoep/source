- hive 데이터 베이스 스키마 (하위 테이블) 복사 shell script

#!/usr/bin/env sh

DATABASE_NM='test'
TABLE_NM=`hive -S -e "use $DATABASE_NM; show tables;" | grep -v "WARN"`

for table in $TABLE_NM
do
    hive -S -e "show create table ${DATABASE_NM}.$table" | grep -v "WARN" >> ${DATABASE_NM}_schema.hql
    echo -e ";\n" >> ${DATABASE_NM}_schema.hql
done



- 데이터 베이스 스키마 전체 추가

hive -f ${DATABASE_NM}_schema.hql
